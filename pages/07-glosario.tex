\makeglossaries

\newglossaryentry{ctf}{
    name        =   CTF,
    description =   {Competición de ciberseguridad, popular entre los profesionales de la seguridad informática y los estudiantes de tecnología de la información, en la que los participantes se enfrentan encontrando y explotando vulnerabilidades en sistemas informáticos y aplicaciones con el objetivo de obtener una "bandera" o conjunto de datos confidenciales.}
}
